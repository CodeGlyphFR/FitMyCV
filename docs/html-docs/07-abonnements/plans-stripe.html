Configuration Stripe

// Modèle SubscriptionPlan
model SubscriptionPlan {
  id                String   @id @default(cuid())
  name              String
  stripePriceId     String   @unique
  stripeProductId   String
  price             Decimal
  interval          String   // monthly, yearly
  features          Json
  limits            Json
  isActive          Boolean  @default(true)
}
