// lib/cv-optimisation/apply.js
export async function applySuggestion(cvFileId, suggestionId, userId) {
  const suggestion = await prisma.cvSuggestion.findUnique({
    where: { id: suggestionId }
  });

  const cvFile = await prisma.cvFile.findFirst({
    where: { id: cvFileId, userId }
  });

  // Appliquer la modification selon le type
  const updatedContent = applyModification(
    cvFile.content,
    suggestion.type,
    suggestion.path,
    suggestion.newValue
  );

  // Sauvegarder avec nouvelle version
  await writeUserCvFile(userId, cvFile.filename, updatedContent, {
    origin: 'ai_optimization',
    description: suggestion.description
  });

  // Marquer la suggestion comme appliquee
  await prisma.cvSuggestion.update({
    where: { id: suggestionId },
    data: { status: 'applied' }
  });
}
