Phase Recomposition

Phase finale du pipeline qui assemble toutes les sections generees en un CV complet et coherent.

Objectif

La recomposition prend les sorties des phases precedentes (classification, batches) et les assemble en respectant :

  • La structure JSON standard du CV
  • L'ordre logique des sections
  • La coherence globale du document

Implementation

// lib/cv-generation/phases/recompose.js
export async function recomposeCv(classificationResult, batchResults, originalCv) {
  const recomposedCv = {
    header: originalCv.header,
    sections: {
      summary: batchResults.summary,
      experiences: filterAndOrder(
        batchResults.experiences,
        classificationResult.experiences
      ),
      skills: batchResults.skills,
      education: originalCv.sections.education,
      projects: mergeProjects(
        originalCv.sections.projects,
        classificationResult.movedToProjects
      )
    },
    metadata: {
      ...originalCv.metadata,
      generatedAt: new Date().toISOString(),
      targetJobId: jobOffer.id
    }
  };

  return recomposedCv;
}
