Export DOCX

Generation de fichiers Word (DOCX) a partir des CV.

Technologies

  • docx : Generation de documents Word
  • Templates : Styles et mise en page

Implementation

// lib/export/docx.js
import { Document, Packer, Paragraph, TextRun } from 'docx';

export async function exportToDocx(cvData) {
  const doc = new Document({
    sections: [{
      children: [
        // Header
        new Paragraph({
          children: [
            new TextRun({
              text: `{cvData.header.firstName} {cvData.header.lastName}`,
              bold: true,
              size: 32
            })
          ]
        }),
        // ... autres sections
      ]
    }]
  });

  const buffer = await Packer.toBuffer(doc);
  return buffer;
}
