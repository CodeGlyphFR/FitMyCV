// lib/subscription/featureUsage.js
export async function canUseFeature(userId, featureName) {
  const plan = await getUserPlan(userId);
  const limit = plan.limits[featureName];

  if (limit === -1) return { allowed: true }; // Illimite

  const counter = await prisma.featureUsageCounter.findFirst({
    where: {
      userId,
      featureName,
      month: getCurrentMonth()
    }
  });

  const used = counter?.count || 0;
  const allowed = used < limit;

  return {
    allowed,
    used,
    limit,
    needsCredit: !allowed && plan.allowCreditOverage
  };
}
