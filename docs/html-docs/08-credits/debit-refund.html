// lib/subscription/credits.js
export async function debitCredit(userId, amount, reason) {
  return prisma.$transaction(async (tx) => {
    const balance = await tx.creditBalance.findUnique({
      where: { userId }
    });

    if (balance.balance < amount) {
      throw new Error('Insufficient credits');
    }

    await tx.creditBalance.update({
      where: { userId },
      data: { balance: { decrement: amount } }
    });

    return tx.creditTransaction.create({
      data: { userId, amount: -amount, type: 'debit', reason }
    });
  });
}

export async function refundCredit(userId, amount, reason) {
  return prisma.$transaction(async (tx) => {
    await tx.creditBalance.update({
      where: { userId },
      data: { balance: { increment: amount } }
    });

    return tx.creditTransaction.create({
      data: { userId, amount, type: 'refund', reason }
    });
  });
}
