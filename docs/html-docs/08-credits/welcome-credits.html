Credits Bienvenue

Attribution automatique de credits gratuits aux nouveaux utilisateurs.

Configuration

Credits offerts

5 credits sont automatiquement attribues a chaque nouvel utilisateur lors de son inscription.

Implementation

// lib/subscription/welcomeCredits.js
const WELCOME_CREDITS = 5;

export async function grantWelcomeCredits(userId) {
  // Verifier si deja attribues
  const existing = await prisma.creditBalance.findUnique({
    where: { userId }
  });

  if (existing) return null;

  // Creer le solde avec les credits de bienvenue
  const balance = await prisma.creditBalance.create({
    data: {
      userId,
      balance: WELCOME_CREDITS
    }
  });

  // Logger la transaction
  await prisma.creditTransaction.create({
    data: {
      userId,
      amount: WELCOME_CREDITS,
      type: 'credit',
      reason: 'welcome_bonus',
      description: 'Credits de bienvenue'
    }
  });

  return balance;
}
