Extraction URL

Extraction automatique des informations d'une offre d'emploi a partir de son URL.

Implementation

// lib/job-offers/extractFromUrl.js
import { fetchPageContent } from './scraper';
import { callOpenAI } from '@/lib/openai-core/client';

export async function extractJobOfferFromUrl(url) {
  // 1. Recuperer le contenu de la page
  const htmlContent = await fetchPageContent(url);

  // 2. Nettoyer le HTML
  const cleanText = cleanHtmlContent(htmlContent);

  // 3. Extraire via IA
  const result = await callOpenAI({
    featureName: 'job_extraction',
    messages: [
      {
        role: 'system',
        content: JOB_EXTRACTION_PROMPT
      },
      {
        role: 'user',
        content: cleanText
      }
    ],
    responseFormat: {
      type: 'json_schema',
      json_schema: jobOfferSchema
    }
  });

  return result;
}

Scraping avec Puppeteer

// lib/job-offers/scraper.js
import puppeteer from 'puppeteer';

export async function fetchPageContent(url) {
  const browser = await puppeteer.launch({
    headless: 'new',
    args: ['--no-sandbox']
  });

  try {
    const page = await browser.newPage();

    // User-Agent realiste
    await page.setUserAgent(
      'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
    );

    await page.goto(url, { waitUntil: 'networkidle0' });

    // Attendre le contenu principal
    await page.waitForSelector('body', { timeout: 10000 });

    return await page.content();
  } finally {
    await browser.close();
  }
}

Gestion des Erreurs

Cas d'erreur
  • Page inaccessible : Retour erreur avec message
  • Contenu bloque : Fallback vers extraction manuelle
  • Format non reconnu : Extraction IA generique
