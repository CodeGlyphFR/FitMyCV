Modèle CreditPack

model CreditPack {
  id              String   @id @default(cuid())
  name            String
  credits         Int
  price           Decimal
  stripePriceId   String   @unique
  isActive        Boolean  @default(true)
  sortOrder       Int      @default(0)
}
