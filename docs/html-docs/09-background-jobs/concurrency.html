Gestion de la Concurrence

Mecanismes de limitation et gestion des taches paralleles.

Limites de Concurrence

Limite globale

Maximum 3 taches simultanees par utilisateur pour eviter la surcharge.

Implementation

// lib/background-jobs/jobQueue.js
const MAX_CONCURRENT_JOBS = 3;

export async function enqueueJob(userId, jobFn) {
  // Verifier les jobs en cours
  const runningJobs = await prisma.backgroundTask.count({
    where: {
      userId,
      status: { in: ['pending', 'running'] }
    }
  });

  if (runningJobs >= MAX_CONCURRENT_JOBS) {
    throw new Error('Too many concurrent jobs');
  }

  // Creer et executer le job
  const task = await prisma.backgroundTask.create({
    data: { userId, status: 'pending' }
  });

  // Execution asynchrone
  setImmediate(async () => {
    try {
      await prisma.backgroundTask.update({
        where: { id: task.id },
        data: { status: 'running', startedAt: new Date() }
      });

      const result = await jobFn();

      await prisma.backgroundTask.update({
        where: { id: task.id },
        data: { status: 'completed', result, completedAt: new Date() }
      });
    } catch (error) {
      await prisma.backgroundTask.update({
        where: { id: task.id },
        data: { status: 'failed', error: error.message }
      });
    }
  });

  return task;
}
