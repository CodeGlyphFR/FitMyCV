Système Email

FitMyCV.io utilise Resend pour l'envoi d'emails transactionnels. Les templates sont stockés en base de données et éditables via l'interface admin.

Modèles de Données

EmailTemplate

model EmailTemplate {
  id          Int       @id @default(autoincrement())
  name        String    @unique  // welcome, password_reset, etc.
  subject     String
  htmlContent String    @db.Text
  textContent String?   @db.Text
  trigger     String?   // Event qui déclenche l'envoi
  isActive    Boolean   @default(true)
  variables   Json?     // Variables disponibles
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  logs        EmailLog[]
}

EmailLog

model EmailLog {
  id          Int       @id @default(autoincrement())
  templateId  Int
  recipient   String
  subject     String
  status      String    // sent, failed, bounced
  resendId    String?   // ID Resend pour tracking
  error       String?
  sentAt      DateTime  @default(now())

  template    EmailTemplate @relation(fields: [templateId], references: [id])
}

Envoi d'Email

// lib/email/sender.js
import { Resend } from 'resend';

const resend = new Resend(process.env.RESEND_API_KEY);

export async function sendEmail(templateName, recipient, variables = {}) {
  // 1. Charger le template
  const template = await prisma.emailTemplate.findUnique({
    where: { name: templateName }
  });

  if (!template || !template.isActive) {
    throw new Error(`Template ${templateName} not found or inactive`);
  }

  // 2. Interpoler les variables
  const subject = interpolate(template.subject, variables);
  const html = interpolate(template.htmlContent, variables);
  const text = template.textContent
    ? interpolate(template.textContent, variables)
    : null;

  // 3. Envoyer via Resend
  try {
    const result = await resend.emails.send({
      from: 'FitMyCV ',
      to: recipient,
      subject,
      html,
      text
    });

    // 4. Logger l'envoi
    await prisma.emailLog.create({
      data: {
        templateId: template.id,
        recipient,
        subject,
        status: 'sent',
        resendId: result.id
      }
    });

    return result;

  } catch (error) {
    // Logger l'échec
    await prisma.emailLog.create({
      data: {
        templateId: template.id,
        recipient,
        subject,
        status: 'failed',
        error: error.message
      }
    });

    throw error;
  }
}

function interpolate(text, variables) {
  return text.replace(/\{\{(\w+)\}\}/g, (_, key) => variables[key] || '');
}

Triggers Automatiques

// lib/email/triggers.js

export async function onUserCreated(user) {
  const welcomeCredits = await getSetting('welcome_credits');

  await sendEmail('welcome', user.email, {
    userName: user.name || user.email,
    welcomeCredits
  });
}

export async function onSubscriptionCreated(userId, subscription) {
  const user = await prisma.user.findUnique({ where: { id: userId } });
  const plan = await prisma.subscriptionPlan.findUnique({
    where: { stripePriceId: subscription.stripePriceId }
  });

  await sendEmail('subscription_created', user.email, {
    userName: user.name,
    planName: plan.displayName,
    price: `${plan.priceMonthly}€/mois`
  });
}

export async function onCvGenerationComplete(userId, cvFile) {
  const user = await prisma.user.findUnique({ where: { id: userId } });

  await sendEmail('cv_generation_complete', user.email, {
    userName: user.name,
    cvName: cvFile.filename,
    dashboardUrl: `${process.env.NEXTAUTH_URL}/dashboard`
  });
}
