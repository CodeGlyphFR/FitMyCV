Modèle de Données

model OnboardingProgress {
  id          Int       @id @default(autoincrement())
  userId      String    @unique
  step        Int       @default(1)
  completed   Boolean   @default(false)
  method      String?   // pdf_import, manual
  startedAt   DateTime  @default(now())
  completedAt DateTime?
  metadata    Json?     // Données additionnelles

  user        User      @relation(fields: [userId], references: [id])
}

Tracking Analytics

// lib/analytics/onboarding.js

export async function trackOnboardingStep(userId, step, metadata = {}) {
  await prisma.onboardingProgress.upsert({
    where: { userId },
    update: {
      step,
      metadata: {
        ...existingMetadata,
        [`step${step}At`]: new Date().toISOString(),
        ...metadata
      }
    },
    create: {
      userId,
      step,
      metadata
    }
  });

  // Envoyer à analytics
  await trackEvent('onboarding_step', {
    userId,
    step,
    ...metadata
  });
}

export async function completeOnboarding(userId) {
  await prisma.onboardingProgress.update({
    where: { userId },
    data: {
      completed: true,
      completedAt: new Date()
    }
  });

  await trackEvent('onboarding_completed', { userId });
}
